alias chols=f_chols
alias chomkdir=f_chomkdir
function chocd {
  if [ $# = 0 -a "$CHODEFHOME" != "" ]; then
    cd "$CHODEFHOME"
  elif [ $# = 1 ]; then
    cd `chomap "$1"`
  else
    echo "usage: chocd [tag]" 1>&2
  fi
}
function chou {
  pushd . > /dev/null
  chocd "$1"
  dirs
}
function f_chomkdir {
  \chomkdir "$1" && chocd a
}
function f_chols {
  \chols "$@" | more
}
